def calc_rsi(nav: pd.Series, period: int = 14) -> pd.Series:
    """计算 RSI 相对强弱指标，返回 0-100 Series"""
    delta = nav.diff()
    gain = delta.clip(lower=0)
    loss = (-delta).clip(lower=0)
    avg_gain = gain.rolling(window=period).mean()
    avg_loss = loss.rolling(window=period).mean()
    rs = avg_gain / avg_loss.replace(0, 1e-10)
    return 100 - (100 / (1 + rs))
```

def calc_bollinger(nav: pd.Series, period: int = 20, std: float = 2.0):
    """计算布林带，返回 (upper, middle, lower) 三个 Series"""
    middle = nav.rolling(window=period).mean()
    rolling_std = nav.rolling(window=period).std()
    upper = middle + std * rolling_std
    lower = middle - std * rolling_std
    return upper, middle, lower
```

def calc_volatility(nav: pd.Series, period: int = 20) -> float:
    """计算年化波动率（最后 period 日标准差 × √252）"""
    returns = nav.pct_change().dropna().iloc[-period:]
    if len(returns) < 2:
        return 0.0
    return float(returns.std() * (252 ** 0.5))

import numpy as np


def extract_feature_vector(fund_nav: pd.Series, benchmark_nav: pd.Series) -> dict:
    """从基金 NAV 和基准 NAV 提取 6 维特征向量，全部归一化到 [0,1]"""

    def _sigmoid(x: float, scale: float = 1.0) -> float:
        return float(1.0 / (1.0 + np.exp(-x * scale)))

    def _clip01(x: float) -> float:
        return float(max(0.0, min(1.0, x)))

    # 计算所需指标
    nav = fund_nav.values
    ma5 = pd.Series(nav).rolling(5).mean()
    ma10 = pd.Series(nav).rolling(10).mean()
    ma20 = pd.Series(nav).rolling(20).mean()
    rsi = calc_rsi(pd.Series(nav), 14)
    upper, middle, lower = calc_bollinger(pd.Series(nav), 20, 2.0)
    vol = calc_volatility(pd.Series(nav), 20)

    # MACD
    ema12 = pd.Series(nav).ewm(span=12, adjust=False).mean()
    ema26 = pd.Series(nav).ewm(span=26, adjust=False).mean()
    dif = ema12 - ema26
    dea = dif.ewm(span=9, adjust=False).mean()

    latest = -1

    # F1: MA 排列得分
    v_ma5 = nav[latest]; v_ma10 = ma10.iloc[latest]; v_ma20 = ma20.iloc[latest]
    if v_ma10 > 1e-10 and v_ma20 > 1e-10:
        a1 = (nav[latest] - ma5.iloc[latest]) / ma5.iloc[latest]
        a2 = (ma5.iloc[latest] - v_ma10) / v_ma10
        a3 = (v_ma10 - v_ma20) / v_ma20
        f1 = _sigmoid((a1 + a2 + a3) / 3, 10.0)
    else:
        f1 = 0.5

    # F2: MACD 动量
    f2 = _sigmoid((dif.iloc[latest] - dea.iloc[latest]) / (nav[latest] + 1e-10) * 100, 1.0)

    # F3: RSI 位置
    rsi_val = rsi.iloc[latest]
    f3 = _clip01((rsi_val - 30.0) / 40.0) if not np.isnan(rsi_val) else 0.5

    # F4: 布林带位置
    band_range = upper.iloc[latest] - lower.iloc[latest]
    if band_range > 1e-10:
        f4 = _clip01((nav[latest] - lower.iloc[latest]) / band_range)
    else:
        f4 = 0.5

    # F5: 年化波动率
    f5 = _clip01(vol / 0.5)

    # F6: 基准相对强弱
    fund_20d_ret = (nav[latest] - nav[max(0, latest - 20)]) / (nav[max(0, latest - 20)] + 1e-10)
    bench_vals = benchmark_nav.values
    bench_20d_ret = (bench_vals[latest] - bench_vals[max(0, latest - 20)]) / (bench_vals[max(0, latest - 20)] + 1e-10)
    f6 = _sigmoid((fund_20d_ret - bench_20d_ret) * 5, 1.0)

    return {
        "F1_ma_align": round(f1, 4),
        "F2_macd_momentum": round(f2, 4),
        "F3_rsi": round(f3, 4),
        "F4_bollinger": round(f4, 4),
        "F5_volatility": round(f5, 4),
        "F6_benchmark": round(f6, 4),
    }
```

import pandas as pd
import numpy as np
from data.technical import extract_feature_vector


def build_feature_matrix(
    fund_nav: pd.Series,
    benchmark_nav: pd.Series,
    lookback: int = 20,
    return_horizons: tuple = (22, 44, 66),
) -> pd.DataFrame:
    """滚动窗口构建特征矩阵 + 前瞻收益。
    
    对每个能满足 lookback 点位的窗口：
    - 用 lookback 天数据提取 6 维特征向量
    - 记录后续 22/44/66 交易日的净值收益
    """
    rows = []
    nav_vals = fund_nav.values

    for i in range(lookback, len(nav_vals)):
        # 确保有足够的前瞻数据
        max_horizon = max(return_horizons)
        if i + max_horizon >= len(nav_vals):
            break

        # 取 [i-lookback, i] 范围的数据
        fund_window = fund_nav.iloc[i - lookback:i + 1].reset_index(drop=True)
        bench_window = benchmark_nav.iloc[i - lookback:i + 1].reset_index(drop=True)

        try:
            features = extract_feature_vector(fund_window, bench_window)
        except Exception:
            continue

        row = {**features}
        for h in return_horizons:
            start_nav = nav_vals[i]
            end_nav = nav_vals[i + h]
            row[f"forward_{h}d"] = float((end_nav - start_nav) / start_nav)

        rows.append(row)

    return pd.DataFrame(rows)
```

from scipy.spatial.distance import cdist


FEATURE_COLS = ["F1_ma_align", "F2_macd_momentum", "F3_rsi", "F4_bollinger", "F5_volatility", "F6_benchmark"]


def find_similar_patterns(
    current_features: dict,
    feature_matrix: pd.DataFrame,
    top_n: int = 30,
) -> pd.DataFrame:
    """用余弦相似度找出与当前特征最相似的 top_n 个历史窗口"""
    current_vec = np.array([[current_features[k] for k in FEATURE_COLS]])
    hist_mat = feature_matrix[FEATURE_COLS].values

    # 余弦距离 → 余弦相似度
    cosine_dist = cdist(current_vec, hist_mat, metric="cosine")[0]
    similarity = 1.0 - cosine_dist

    result = feature_matrix.copy()
    result["similarity"] = similarity
    return result.nlargest(top_n, "similarity")
```

def compute_period_prediction(forward_returns: pd.Series) -> Optional[PeriodPrediction]:
    """从一组历史前瞻收益中统计预测指标"""
    if len(forward_returns) < 5:
        return None

    returns = forward_returns.dropna()
    if len(returns) < 5:
        return None

    positive_mask = returns > 0
    win_prob = float(positive_mask.sum() / len(returns))

    return PeriodPrediction(
        win_probability=round(win_prob, 2),

import numpy as np


def predict_fund_return(
    fund_nav: pd.Series,
    benchmark_nav: pd.Series,
    code: str = "",
    name: str = "",
    lookback: int = 20,
    top_n: int = 30,
) -> PredictionResult:
    """主预测函数：输入基金NAV和基准NAV，返回 PredictionResult"""
    if len(fund_nav) < 60:
        return PredictionResult(
            code=code, name=name,
            current_features={},
            match_count=0, avg_similarity=0.0,
            pred_1m=None, pred_2m=None, pred_3m=None,
            confidence="低",
        )

    # 1. 当前特征
    current_features = extract_feature_vector(
        fund_nav.iloc[-lookback - 1:].reset_index(drop=True),
        benchmark_nav.iloc[-lookback - 1:].reset_index(drop=True),
    )

    # 2. 特征矩阵
    feat_matrix = build_feature_matrix(fund_nav, benchmark_nav, lookback=lookback)

    if feat_matrix.empty or len(feat_matrix) < 5:
        return PredictionResult(
            code=code, name=name,
            current_features=current_features,
            match_count=0, avg_similarity=0.0,
            pred_1m=None, pred_2m=None, pred_3m=None,
            confidence="低",
        )

    # 3. 相似度匹配（排除自身 — 移除最相似的 1 行）
    matches = find_similar_patterns(current_features, feat_matrix, top_n=top_n + 1)
    if len(matches) > 1:
        matches = matches.iloc[1:]  # 去掉自身
    elif len(matches) > 0:
        matches = matches.iloc[1:] if matches.iloc[0]["similarity"] > 0.99 else matches

    avg_sim = float(matches["similarity"].mean())

    # 4. 统计预测
    pred_1m = compute_period_prediction(matches["forward_22d"])
    pred_2m = compute_period_prediction(matches["forward_44d"])
    pred_3m = compute_period_prediction(matches["forward_66d"])

    # 5. 置信度
    if avg_sim >= 0.75:
        confidence = "高"
    elif avg_sim >= 0.50:
        confidence = "中"
    else:
        confidence = "低"

    return PredictionResult(
        code=code, name=name,
        current_features=current_features,
        match_count=len(matches),
        avg_similarity=round(avg_sim, 4),
        pred_1m=pred_1m,
        pred_2m=pred_2m,
        pred_3m=pred_3m,
        confidence=confidence,
    )

def plot_return_distribution(
    pred_1m: object,  # PeriodPrediction
    pred_2m: object,
    pred_3m: object,
    fund_name: str,
) -> go.Figure:
    """绘制三个周期的收益分布对比直方图"""
    from plotly.subplots import make_subplots

    fig = make_subplots(
        rows=1, cols=3,
        subplot_titles=("近1月", "近2月", "近3月"),
        shared_yaxes=True,
    )

    periods = [
        (pred_1m, "1月", 0),
        (pred_2m, "2月", 1),
        (pred_3m, "3月", 2),
    ]

    for pred, label, col in periods:
        if pred is None:
            continue
        # 构造正态近似柱状数据
        x_vals = [pred.max_loss, pred.p25_return, pred.median_return, pred.p75_return, pred.max_gain]
        x_labels = ["最大亏损", "P25", "中位", "P75", "最大收益"]
        colors_bar = ["#ef5350" if v < 0 else "#4caf50" for v in x_vals]

        fig.add_trace(
            go.Bar(
                x=x_labels, y=x_vals,
                marker_color=colors_bar,
                name=f"{label}",
                text=[f"{v:+.1%}" for v in x_vals],
                textposition="outside",
            ),
            row=1, col=col + 1,
        )
        fig.add_hline(y=0, line_dash="dash", line_color="gray", row=1, col=col + 1)

    fig.update_layout(
        title=f"{fund_name} — 盈利预测分布",
        height=300,
        showlegend=False,
        margin=dict(l=20, r=20, t=50, b=20),
    )
    return fig
```
